const express = require("express");


const app = express();


app.use(express.json());


let orders = [];


app.get("/", (req, res) => {

  res.send("Fixar server is running");

});


app.post("/order", (req, res) => {


  const order = req.body;


  orders.push(order);


  console.log("New order:", order);


  res.json({

    message: "Order received"

  });


});


app.get("/orders", (req, res) => {


  res.json(orders);


});


app.listen(3000, () => {

  console.log("Fixar server running on port 3000");

});
